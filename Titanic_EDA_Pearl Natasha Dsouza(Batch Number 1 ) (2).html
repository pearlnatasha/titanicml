def barchart(feature):
    survived=eda[eda['Survived']==1][feature].value_counts()
    dead=eda[eda['Survived']==0][feature].value_counts()
    a2=pd.DataFrame([survived,dead])
    a2.index=['Survived', 'Dead']
    a2.plot(kind='bar',figsize=(6,5))
    return a2
